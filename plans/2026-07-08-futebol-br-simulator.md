# Futebol BR Simulator Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a deterministic JavaScript simulator that produces one full season of the Brazilian Football 2.0 reform (6 Regional Leagues, Copa dos Campeões, Copa do Brasil, Conmebol slot allocation, per-club calendars) and bakes the result to `data/season-seed-0.json` for consumption by the frontend site.

**Architecture:** Pure ES modules, isomorphic (same code runs in Node for baking and in the browser for future reroll). Seeded RNG for reproducibility. Poisson goal model derived from a ranking differential. Denormalized/render-ready JSON output (pages become dumb templates in Plan 2). Vitest + strict TDD for all sim logic. No dependencies at runtime except a tiny RNG primitive we write ourselves.

**Tech Stack:**
- Node ≥ 20 (native ESM)
- Vitest for testing
- `package.json` `"type": "module"`
- Zero runtime deps

**Key rules the sim enforces (source: `initial-data/regras-negocio.md`, discussion decisions):**
- 192 clubs total: 108 Série A (18 × 6 regiões) + 84 Série B (14 × 6 regiões).
- Season window: late Jan → early Dec, 42 usable weeks after FIFA + mid-season pause.
- Match model: Poisson goals with expected value = f(ranking differential).
- Regional Leagues: 34 rodadas turno+returno, exclusivamente aos fins de semana.
- Copa dos Campeões: 48 clubs, 4-pot geo-locked draw, 12 grupos × 4, 24 top-2 + 8 best 3rds → 32 KO, jogo único, midweeks only.
- Copa do Brasil: 144 clubs, elite 13 bypass to 16-avos, funil 26→13→59→30→15→(+4 Lucky Losers)→19 + 13 elite = 32, jogo único, midweeks only. Pairing: seeded early → open draw at 16-avos → fixed bracket to final.
- Conmebol: Libertadores 7 vagas hierárquicas (cascade), Sul-Am 6 vagas (uma por região).
- Elite ceiling: 60 jogos. Base ceiling: 34.
- Deterministic under seed. Same seed → same season, byte-for-byte JSON.

// src/sim/rng.js
// Mulberry32: fast, deterministic, small-period PRNG. Fine for gameplay-scale sims.
export function createRng(seed) {
  let s = seed >>> 0;
  return function rng() {
    s = (s + 0x6D2B79F5) >>> 0;
    let t = s;
    t = Math.imul(t ^ (t >>> 15), t | 1);
    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  };
}
```

// src/sim/poisson.js
// Knuth's inverse-transform Poisson sampling. Fine for small lambda (<20).
export function samplePoisson(lambda, rng) {
  if (lambda <= 0) return 0;
  const L = Math.exp(-lambda);
  let k = 0;
  let p = 1;
  do {
    k++;
    p *= rng();
  } while (p > L);
  return k - 1;
}
```

**Files:**

**Design notes to bake into the code:**
- `ranking_forca` is on a 1–10 scale.
- Base expected goals per side: `1.3` (roughly average Brazilian league goals per team per match).
- Adjust by ranking differential: each 1-point advantage in ranking shifts expected goals by `+0.15` for the stronger side (`-0.15` for the weaker). Small home-side bonus: `+0.20` to the home team's expected goals.
- Clamp lambda to `[0.15, 5]` to avoid degenerate values.

// src/sim/match.js
import { samplePoisson } from './poisson.js';

const BASE_LAMBDA = 1.3;
const RANKING_STEP = 0.15;
const HOME_BONUS = 0.20;
const MIN_LAMBDA = 0.15;
const MAX_LAMBDA = 5;

function clamp(x, lo, hi) {
  return Math.min(hi, Math.max(lo, x));
}

export function simulateMatch({ rankingHome, rankingAway }, rng) {
  const diff = rankingHome - rankingAway;
  const lambdaHome = clamp(BASE_LAMBDA + RANKING_STEP * diff + HOME_BONUS, MIN_LAMBDA, MAX_LAMBDA);
  const lambdaAway = clamp(BASE_LAMBDA - RANKING_STEP * diff, MIN_LAMBDA, MAX_LAMBDA);
  return {
    golsCasa: samplePoisson(lambdaHome, rng),
    golsFora: samplePoisson(lambdaAway, rng),
  };
}
```

// src/data/teams.js
import { readFileSync } from 'node:fs';
import { fileURLToPath } from 'node:url';
import { dirname, resolve } from 'node:path';
import { createRng } from '../sim/rng.js';

const __dirname = dirname(fileURLToPath(import.meta.url));
const RAW_PATH = resolve(__dirname, '../../initial-data/teams.json');

// Canonical 3-letter IDs for well-known clubs. Anything not in this map falls back to a derived ID.
const CANONICAL_IDS = {
  'Palmeiras': 'PAL', 'São Paulo': 'SAO', 'Corinthians': 'COR', 'Santos': 'SAN',
  'Red Bull Bragantino': 'RBB', 'Novorizontino': 'NOV', 'Mirassol': 'MIR',
  'Flamengo': 'FLA', 'Botafogo': 'BOT', 'Fluminense': 'FLU', 'Vasco da Gama': 'VAS',
  'Fortaleza': 'FOR', 'Bahia': 'BAH', 'Ceará': 'CEA', 'Vitória': 'VIT', 'Sport Recife': 'SPT',
  'Internacional': 'INT', 'Grêmio': 'GRE', 'Athletico-Paranaense': 'CAP',
  'Cruzeiro': 'CRU', 'Atlético-MG': 'CAM', 'América-MG': 'AMG', 'Goiás': 'GOI',
};

function deriveId(name, taken) {
  const clean = name.toUpperCase().replace(/[^A-Z0-9]/g, '');
  let base = clean.slice(0, 3);
  if (!taken.has(base)) return base;
  for (let i = 1; i < 100; i++) {
    const alt = base.slice(0, 2) + i.toString(36).toUpperCase();
    if (!taken.has(alt)) return alt;
  }
  throw new Error(`Cannot derive unique ID for ${name}`);
}

function seededSerieB(ligaName, count) {
  // Deterministic pseudo-hash from the league name to seed placeholder rankings.
  let seed = 0;
  for (const c of ligaName) seed = ((seed * 31) + c.charCodeAt(0)) >>> 0;
  const rng = createRng(seed);
  const clubes = [];
  for (let i = 1; i <= count; i++) {
    const num = String(i).padStart(2, '0');
    const ranking = 1 + Math.floor(rng() * 4); // 1..4
    clubes.push({
      nome: `${ligaName.replace(/^Liga /, '')} B-${num}`,
      estado: '—',
      ranking_forca: ranking,
      divisao: 'B',
    });
  }
  return clubes;
}

function normalizeLigaKey(nome) {
  // "Liga Paulista" → "Liga Paulista" (already canonical); handle "Liga Sulista (RS, SC, PR)" if it appears.
  return nome.split('(')[0].trim();
}

let _ligas = null;
let _byId = null;

function build() {
  const raw = JSON.parse(readFileSync(RAW_PATH, 'utf8'));
  const taken = new Set();
  const ligas = raw.ligas.map((liga) => {
    const nome = normalizeLigaKey(liga.nome);
    const serieA = liga.clubes.map((c) => {
      const id = CANONICAL_IDS[c.nome] ?? deriveId(c.nome, taken);
      taken.add(id);
      return {
        id,
        nome: c.nome,
        estado: c.estado,
        ranking_forca: c.ranking,
        divisao: 'A',
        liga_regional: nome,
      };
    });
    const serieB = seededSerieB(nome, 14).map((c) => {
      const id = deriveId(c.nome, taken);
      taken.add(id);
      return { ...c, id, liga_regional: nome };
    });
    return { nome, clubes: [...serieA, ...serieB] };
  });
  const byId = new Map();
  for (const l of ligas) for (const c of l.clubes) byId.set(c.id, c);
  return { ligas, byId };
}

function ensure() {
  if (!_ligas) {
    const { ligas, byId } = build();
    _ligas = ligas;
    _byId = byId;
  }
}

export function getLeagues() {
  ensure();
  return _ligas;
}
export function getAllTeams() {
  ensure();
  return _ligas.flatMap((l) => l.clubes);
}
export function getTeamById(id) {
  ensure();
  return _byId.get(id);
}
```

// src/sim/regional-league.js
// Circle method: fix team 0, rotate the rest. Then generate the returno by swapping home/away.
export function buildSchedule(teams) {
  const n = teams.length;
  if (n % 2 !== 0) throw new Error('buildSchedule requires an even number of teams');
  const rotation = teams.slice();
  const rounds = [];
  const half = n / 2;
  for (let r = 0; r < n - 1; r++) {
    const round = [];
    for (let i = 0; i < half; i++) {
      const home = rotation[i];
      const away = rotation[n - 1 - i];
      // alternate home/away by round parity for a lightly-balanced schedule
      if ((r + i) % 2 === 0) round.push({ casaId: home.id, foraId: away.id });
      else round.push({ casaId: away.id, foraId: home.id });
    }
    rounds.push(round);
    // rotate: keep first fixed, move last to position 1
    rotation.splice(1, 0, rotation.pop());
  }
  // Returno: same pairings, swapped home/away
  const returno = rounds.map((round) => round.map((m) => ({ casaId: m.foraId, foraId: m.casaId })));
  return [...rounds, ...returno];
}

function shuffle(arr, rng) {
  const a = arr.slice();
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(rng() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

export function drawGroups(pots, ligaLookup, rng) {
  const MAX_ATTEMPTS = 200;
  for (let attempt = 0; attempt < MAX_ATTEMPTS; attempt++) {
    const groups = Array.from({ length: 12 }, () => []);
    const shuffledPots = pots.map((pot) => shuffle(pot, rng));
    let ok = true;
    for (let p = 0; p < 4 && ok; p++) {
      const pot = shuffledPots[p];
      // For each club in this pot, try to place it into the first group without a same-region conflict
      const remaining = pot.slice();
      const groupOrder = shuffle(
        Array.from({ length: 12 }, (_, i) => i),
        rng
      );
      const placed = new Array(12).fill(false);
      for (const clubId of remaining) {
        let placedThis = false;
        for (const gi of groupOrder) {
          if (placed[gi]) continue;
          const region = ligaLookup.get(clubId);
          const conflict = groups[gi].some((c) => ligaLookup.get(c.id) === region);
          if (conflict) continue;
          groups[gi].push({ id: clubId, pot: p + 1 });
          placed[gi] = true;
          placedThis = true;
          break;
        }
        if (!placedThis) { ok = false; break; }
      }
    }
    if (ok) return groups;
  }
  throw new Error('drawGroups: could not satisfy geo-lock after retries');
}

function emptyGroupRow(id) {
  const t = getTeamById(id);
  return {
    id, nome: t?.nome ?? id,
    jogos: 0, vitorias: 0, empates: 0, derrotas: 0,
    golsPro: 0, golsContra: 0, saldoGols: 0, pontos: 0,
  };
}

function applyGroupResult(rowA, rowB, gA, gB) {
  rowA.jogos++; rowB.jogos++;
  rowA.golsPro += gA; rowA.golsContra += gB;
  rowB.golsPro += gB; rowB.golsContra += gA;
  rowA.saldoGols = rowA.golsPro - rowA.golsContra;
  rowB.saldoGols = rowB.golsPro - rowB.golsContra;
  if (gA > gB) { rowA.vitorias++; rowB.derrotas++; rowA.pontos += 3; }
  else if (gA < gB) { rowB.vitorias++; rowA.derrotas++; rowB.pontos += 3; }
  else { rowA.empates++; rowB.empates++; rowA.pontos++; rowB.pontos++; }
}

function sortGroupTable(a, b) {
  if (b.pontos !== a.pontos) return b.pontos - a.pontos;
  if (b.saldoGols !== a.saldoGols) return b.saldoGols - a.saldoGols;
  if (b.golsPro !== a.golsPro) return b.golsPro - a.golsPro;
  return a.nome.localeCompare(b.nome);
}

const GROUP_LABELS = 'ABCDEFGHIJKL';

// Proper round-robin for a group of 4 (3 rounds of 2 matches each).
// Round 1: A-B, C-D. Round 2: A-C, D-B. Round 3: A-D, B-C.
const GROUP_ROUNDS = [
  [[0, 1], [2, 3]],
  [[0, 2], [3, 1]],
  [[0, 3], [1, 2]],
];

export function simulateGroupStage(groups, rng) {
  const grupos = groups.map((g, gi) => {
    const rows = new Map(g.map((c) => [c.id, emptyGroupRow(c.id)]));
    const jogos = [];
    GROUP_ROUNDS.forEach((round, rIdx) => {
      for (const [hi, ai] of round) {
        const home = g[hi], away = g[ai];
        const homeTeam = getTeamById(home.id);
        const awayTeam = getTeamById(away.id);
        const { golsCasa, golsFora } = simulateMatch(
          { rankingHome: homeTeam.ranking_forca, rankingAway: awayTeam.ranking_forca },
          rng
        );
        applyGroupResult(rows.get(home.id), rows.get(away.id), golsCasa, golsFora);
        jogos.push({
          rodada: `grupos-${rIdx + 1}`,
          casaId: home.id, foraId: away.id,
          golsCasa, golsFora,
        });

    return {

      pots: Object.fromEntries(g.map((c) => [c.id, c.pot])),

  });

    golsCasa += extra.golsCasa;
    golsFora += extra.golsFora;
  }
  let vencedorId;
  if (golsCasa > golsFora) vencedorId = homeId;
  else if (golsCasa < golsFora) vencedorId = awayId;
  else {
    // Coin flip weighted very lightly by ranking
    const homeChance = 0.5 + (home.ranking_forca - away.ranking_forca) * 0.02;
    vencedorId = rng() < homeChance ? homeId : awayId;
  }
  return { casaId: homeId, foraId: awayId, golsCasa, golsFora, vencedorId };
}

export function simulateKnockout(clubes32, rng) {
  const shuffled = shuffle(clubes32.map((c) => c.id), rng);
  const rounds = { '16avos': [], oitavas: [], quartas: [], semis: [], final: null };
  let current = shuffled;
  for (const key of ['16avos', 'oitavas', 'quartas', 'semis']) {
    const next = [];
    for (let i = 0; i < current.length; i += 2) {
      const match = playKnockoutMatch(current[i], current[i + 1], rng);
      rounds[key].push(match);
      next.push(match.vencedorId);
    }
    current = next;
  }
  const finalMatch = playKnockoutMatch(current[0], current[1], rng);
  rounds.final = finalMatch;
  rounds.campeao = finalMatch.vencedorId;
  rounds.vice = finalMatch.vencedorId === finalMatch.casaId ? finalMatch.foraId : finalMatch.casaId;
  // Also expose semifinalistas (losers of quartas' winners = winners of semis are the finalistas, so semifinalists = losers of semis)
  const semisLosers = rounds.semis.map((m) => (m.vencedorId === m.casaId ? m.foraId : m.casaId));
  rounds.semifinalistas = semisLosers;
  return rounds;
}

function shuffle(arr, rng) {
  const a = arr.slice();
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(rng() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

// src/sim/conmebol.js
export function allocateConmebolSlots({ copaCampeoes, copaBrasil, ligasRegionais }) {
  const libertadores = [];
  const seen = new Set();
  const cascadePool = copaCampeoes.campanhaGeral.slice();

  function tryAdd(id) {
    if (id && !seen.has(id)) {
      libertadores.push(id);
      seen.add(id);
      return true;
    }
    return false;
  }
  function cascade() {
    while (cascadePool.length) {
      const next = cascadePool.shift();
      if (tryAdd(next)) return;
    }
  }

  const candidates = [
    copaCampeoes.campeao,
    copaCampeoes.vice,
    copaBrasil.campeao,
    copaCampeoes.semifinalistas?.[0],
    copaCampeoes.semifinalistas?.[1],
    copaBrasil.vice,
  ];
  for (const id of candidates) {
    if (!tryAdd(id)) cascade();
  }
  // Slot 7: G7 (melhor campanha geral não classificado)
  cascade();
  while (libertadores.length < 7) cascade();

  const sulAmericana = [];
  for (const liga of ligasRegionais) {
    // Best-placed non-Libertadores club in this liga's tabelaA
    const pick = liga.tabelaA.find((row) => !seen.has(row.id));
    if (pick) {
      sulAmericana.push(pick.id);
      seen.add(pick.id);
    }
  }

  return { libertadores, sulAmericana };
}

// src/sim/calendar.js
export const TOTAL_WEEKS = 42;

// Regional League: rounds 1..34 → weeks 1..34.
// Copa dos Campeões: 3 group rounds + 5 KO rounds = 8 midweeks.
// Copa do Brasil: 4 funnel + 5 KO = 9 midweeks.
// FIFA dates: 4 midweeks. Mid-season break: 1 weekend + 1 midweek (paused entirely).
// Chosen schedule:
const LIGA_WEEKS = Array.from({ length: 34 }, (_, i) => i + 1);
const COPA_CAMPEOES_WEEKS = {
  'grupos-1': 5, 'grupos-2': 8, 'grupos-3': 11,
  '16avos': 20, 'oitavas': 25, 'quartas': 30, 'semis': 35, 'final': 40,
};
const COPA_BRASIL_WEEKS = {
  'preliminar': 3, 'primeira': 6, 'segunda': 9, 'terceira': 12,
  '16avos': 21, 'oitavas': 26, 'quartas': 31, 'semis': 36, 'final': 41,
};
// FIFA breaks: no midweek matches in weeks 14, 24, 34
const FIFA_MIDWEEKS = new Set([14, 24, 34]);

function emptyWeek() {
  return { fimDeSemana: null, meioDeSemana: null };
}

function assignSlot(clubCal, clubId, semana, slot, entry) {
  const wk = clubCal[semana - 1];
  if (wk[slot] !== null) {
    // Collision: defer to next available week of the same slot type
    return false;
  }
  wk[slot] = entry;
  return true;
}

export function assignCalendar(simResult) {
  const { regionalLeagueMatches, copaCampeoesMatches, copaBrasilMatches, clubIds } = simResult;
  const calendariosPorClube = new Map();
  for (const id of clubIds) {
    calendariosPorClube.set(id, Array.from({ length: TOTAL_WEEKS }, () => emptyWeek()));
  }

  const matchesGeral = [];

  function place(competicao, semana, slotKey, match) {
    const home = calendariosPorClube.get(match.casaId);
    const away = calendariosPorClube.get(match.foraId);
    if (!home || !away) return;
    const entry = { competicao, rodada: match.rodada ?? null, casaId: match.casaId, foraId: match.foraId, golsCasa: match.golsCasa, golsFora: match.golsFora };
    const entryHome = { ...entry, casa: true, adversarioId: match.foraId, golsPro: match.golsCasa, golsContra: match.golsFora };
    const entryAway = { ...entry, casa: false, adversarioId: match.casaId, golsPro: match.golsFora, golsContra: match.golsCasa };
    let attempt = semana;
    while (attempt <= TOTAL_WEEKS) {
      if (slotKey === 'meioDeSemana' && FIFA_MIDWEEKS.has(attempt)) { attempt++; continue; }
      const okHome = home[attempt - 1][slotKey] === null;
      const okAway = away[attempt - 1][slotKey] === null;
      if (okHome && okAway) {
        home[attempt - 1][slotKey] = entryHome;
        away[attempt - 1][slotKey] = entryAway;
        matchesGeral.push({ ...entry, semana: attempt });
        return;
      }
      attempt++;
    }
  }

  // Regional League
  for (const m of regionalLeagueMatches) {
    const semana = LIGA_WEEKS[m.rodada - 1];
    place('liga_regional', semana, 'fimDeSemana', m);
  }
  // Copa dos Campeões
  for (const m of copaCampeoesMatches) {
    const rodadaKey = m.rodada?.startsWith('grupos') ? m.rodada : m.rodada;
    const semana = COPA_CAMPEOES_WEEKS[rodadaKey] ?? 20;
    place('copa_campeoes', semana, 'meioDeSemana', m);
  }
  // Copa do Brasil
  for (const m of copaBrasilMatches) {
    const semana = COPA_BRASIL_WEEKS[m.rodada] ?? 15;
    place('copa_brasil', semana, 'meioDeSemana', m);
  }

  const calendariosPorClubeObj = Object.fromEntries(calendariosPorClube);
  return { calendariosPorClube: calendariosPorClubeObj, matchesGeral };
}

// src/sim/perfis.js
export function computePerfis({ clubIds, jogosPorClube }) {
  const sorted = clubIds
    .slice()
    .sort((a, b) => (jogosPorClube.get(b) ?? 0) - (jogosPorClube.get(a) ?? 0));
  const elite = sorted.slice(0, 24);
  const media = sorted.slice(24, 108);
  const base = sorted.slice(108);
  const avg = (ids) => ids.reduce((s, id) => s + (jogosPorClube.get(id) ?? 0), 0) / ids.length;
  return {
    elite: { count: elite.length, mediaJogos: Number(avg(elite).toFixed(1)), exemplos: elite.slice(0, 3) },
    media: { count: media.length, mediaJogos: Number(avg(media).toFixed(1)), exemplos: media.slice(0, 3) },
    base: { count: base.length, mediaJogos: Number(avg(base).toFixed(1)), exemplos: base.slice(0, 3) },
  };
}

import { seedYearZero } from './seeding.js';
import { simulateAllRegionalLeagues } from './regional-league.js';
import { buildPots, drawGroups, simulateGroupStage, selectKnockoutQualifiers, simulateKnockout } from './copa-campeoes.js';
import { buildParticipantPool, simulateFunnel, simulateMataMata } from './copa-brasil.js';

const META = {
  totalClubes: 192,
  tetoJogos: 60,
  mesesGarantidos: 10,
  desempregadosEmAbril: 0,
  reducaoVoos: '>50%',
  semanasUtilizaveis: 42,
  semanasRodadaDupla: 21,
  semanasRodadaSimples: 21,

function ligaLookupFrom(qualificados) {
  const map = new Map();
  for (const l of qualificados) for (const id of l.qualificadosCampeoes) map.set(id, l.nome);
  return map;
}

function campanhaGeral(grupos) {
  // Aggregate every group-stage row across all groups, sort by pontos > SG > GP > name
  const rows = grupos.flatMap((g) => g.tabela);
  return rows
    .slice()
    .sort((a, b) => {
      if (b.pontos !== a.pontos) return b.pontos - a.pontos;
      if (b.saldoGols !== a.saldoGols) return b.saldoGols - a.saldoGols;
      if (b.golsPro !== a.golsPro) return b.golsPro - a.golsPro;
      return a.nome.localeCompare(b.nome);
    })
    .map((r) => r.id);
}

export function simulateSeason(seed) {
  const rng = createRng(seed);
  const seeded = seedYearZero();

  // Regional Leagues
  const regional = simulateAllRegionalLeagues({ rng });

  // Copa dos Campeões
  const pots = buildPots(regional.ligas);
  const lookup = ligaLookupFrom(regional.ligas);
  const groupsDrawn = drawGroups(pots, lookup, rng);
  const { grupos } = simulateGroupStage(groupsDrawn, rng);
  const { top2, melhoresTerceiros } = selectKnockoutQualifiers(grupos);
  const clubes32Campeoes = [...top2.map((r) => ({ id: r.id })), ...melhoresTerceiros.map((r) => ({ id: r.id }))];
  const koCampeoes = simulateKnockout(clubes32Campeoes, rng);

  // Copa do Brasil
  const pool = buildParticipantPool(seeded);
  const funnel = simulateFunnel(pool, rng);
  const koBrasil = simulateMataMata(pool.eliteBypass, funnel.qualificadosParaMataMata, rng);

  // Conmebol
  const conmebol = allocateConmebolSlots({
    copaCampeoes: {
      campeao: koCampeoes.campeao,
      vice: koCampeoes.vice,
      semifinalistas: koCampeoes.semifinalistas,
      campanhaGeral: campanhaGeral(grupos),
    },
    copaBrasil: { campeao: koBrasil.campeao, vice: koBrasil.vice },
    ligasRegionais: regional.ligas,
  });

  // Calendar
  const regionalMatches = regional.ligas.flatMap((l) => l.matches);
  const campeoesMatches = [
    ...grupos.flatMap((g) => g.jogos), // rodada tagged at generation time
    ...koCampeoes['16avos'].map((m) => ({ ...m, rodada: '16avos' })),
    ...koCampeoes.oitavas.map((m) => ({ ...m, rodada: 'oitavas' })),
    ...koCampeoes.quartas.map((m) => ({ ...m, rodada: 'quartas' })),
    ...koCampeoes.semis.map((m) => ({ ...m, rodada: 'semis' })),
    { ...koCampeoes.final, rodada: 'final' },
  ];
  const brasilMatches = [
    ...funnel.preliminar.matches,
    ...funnel.primeira.matches,
    ...funnel.segunda.matches,
    ...funnel.terceira.matches,
    ...koBrasil['16avos'],
    ...koBrasil.oitavas,
    ...koBrasil.quartas,
    ...koBrasil.semis,
    koBrasil.final,

  const clubIds = getAllTeams().map((t) => t.id);
  const calendar = assignCalendar({
    regionalLeagueMatches: regionalMatches,
    copaCampeoesMatches: campeoesMatches,
    copaBrasilMatches: brasilMatches,
    clubIds,
  });

  // Perfis dashboard
  const jogosPorClube = new Map();
  for (const id of clubIds) jogosPorClube.set(id, 0);
  for (const m of calendar.matchesGeral) {
    jogosPorClube.set(m.casaId, (jogosPorClube.get(m.casaId) ?? 0) + 1);
    jogosPorClube.set(m.foraId, (jogosPorClube.get(m.foraId) ?? 0) + 1);
  }

}
